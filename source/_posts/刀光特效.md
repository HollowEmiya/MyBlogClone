# 刀光特效

## shader

~~~hlsl
Shader "MyShader/Dao Guang"
{
    Properties
    {
        //_CopyColorRT ("Main Texture", 2D) = "white" {}
        _MainTex("Main Tex", 2D) = "white" {}
        [HDR]_MainColor("Main Color", Color) = (1,1,1,1)
        _Alpha("Alpha Val", Range(0,1)) = 0.1
        _FlowVal("Flow Val", Range(0.0,1)) = 0
        _FadeInTime("Fade In Time", Range(0,0.5)) = 0
        _FadeOutKey("Fade Out Key", Range(0.0,1)) = 1
        _FadeOutTime("Fade Out Time", Range(0,1)) = 0.1
        [HDR]_FlowSpecularColor("Flow Specular Color", Color) = (1,1,1,1)
        [HDR]_NoiseColor("Noise Color", Color) = (1,1,1,1)
        [Toggle]_ReverseDir("Reverse Dir", Int) = 0
        
        [Header(Option)]
        [Enum(UnityEngine.Rendering.BlendOp)]  _BlendOp  ("BlendOp", Float) = 0
        [Enum(UnityEngine.Rendering.BlendMode)] _SrcBlend ("SrcBlend", Float) = 1
        [Enum(UnityEngine.Rendering.BlendMode)] _DstBlend ("DstBlend", Float) = 0
        [Enum(Off, 0, On, 1)]_ZWriteMode ("ZWriteMode", float) = 1
        [Enum(UnityEngine.Rendering.CullMode)]_CullMode ("CullMode", float) = 2
        [Enum(UnityEngine.Rendering.CompareFunction)]_ZTestMode ("ZTestMode", Float) = 4
        //[Enum(UnityEngine.Rendering.ColorWriteMask)]_ColorMask ("ColorMask", Float) = 15

        //[Header(Stencil)]
        //[Enum(UnityEngine.Rendering.CompareFunction)]_StencilComp ("Stencil Comparison", Float) = 8
        //[IntRange]_StencilWriteMask ("Stencil Write Mask", Range(0,255)) = 255
        //[IntRange]_StencilReadMask ("Stencil Read Mask", Range(0,255)) = 255
        //[IntRange]_Stencil ("Stencil ID", Range(0,255)) = 0
        //[Enum(UnityEngine.Rendering.StencilOp)]_StencilPass ("Stencil Pass", Float) = 0
        //[Enum(UnityEngine.Rendering.StencilOp)]_StencilFail ("Stencil Fail", Float) = 0
        //[Enum(UnityEngine.Rendering.StencilOp)]_StencilZFail ("Stencil ZFail", Float) = 0
    }
    SubShader
    {
        Tags
        { 
            "RenderType"="Transparent"
            "Queue" = "Transparent"
        }
        
        ZWrite [_ZWriteMode]
        ZTest [_ZTestMode]
        BlendOp [_BlendOp]
        Blend [_SrcBlend] [_DstBlend]
        Pass
        {
            Name "Alpha Clip"
            Tags 
            { 
                //"LightMode" = "UniversalForward"
                "LightMode" = "UniversalForward"
            }

            HLSLPROGRAM
            #pragma vertex vert
            #pragma fragment frag

            #include "Packages/com.unity.render-pipelines.universal/ShaderLibrary/Core.hlsl"

            struct appdata
            {
                float4 vertex : POSITION;
                float2 uv : TEXCOORD0;
            };

            struct v2f
            {
                float2 uv : TEXCOORD0;
                float4 vertex : SV_POSITION;
            };

            //SAMPLER(sampler_ScreenOffRT);
            //TEXTURE2D(_ScreenOffRT);
            //float4 _ScreenOffRT_ST;

            CBUFFER_START(UnityPerMaterial)
                SAMPLER(sampler_MainTex);
                TEXTURE2D(_MainTex);
                float4 _MainTex_ST, _MainColor;
                float4 _FlowSpecularColor, _NoiseColor;
                float _Alpha, _FlowVal, _FadeInTime;
                float _FadeOutKey, _FadeOutTime;
                int _ReverseDir;
            CBUFFER_END

            v2f vert (appdata v)
            {
                v2f o;
                o.vertex = mul(UNITY_MATRIX_MVP,v.vertex);
                o.uv = TRANSFORM_TEX(v.uv,_MainTex);
                return o;
            }

            half4 frag (v2f i) : SV_Target
            {
                // sample the texture
                float flowUV = i.uv.x;
                if(_ReverseDir != 0)
                    flowUV = 1 - flowUV;
                //float val = i.uv.x;
                flowUV = flowUV + _FlowVal;
                float alphaFade = clamp((_FlowVal / _FadeInTime), 0 ,1);
                //flowUV *= _FlowVal;

                half4 mainCol = SAMPLE_TEXTURE2D(_MainTex, sampler_MainTex, float2(flowUV,i.uv.y));
                float alpha = alphaFade * mainCol.a;
                alpha *= 1 - clamp( (_FlowVal - _FadeOutKey) / _FadeOutTime, 0, 1);
                clip(_Alpha - mainCol.a);
                
                float3 col = _FlowSpecularColor * mainCol.g;
                col += mainCol.r * _MainColor;
                col += _NoiseColor * mainCol.b;
                return half4(col, alpha);
                //return col * flowUV * alphaValByFlow;// * _FlowSpecularColor;
            }

            ENDHLSL
        }
    }
}

~~~

主要思路就根据 uv.x 做渐入渐出和透明度的调整。
